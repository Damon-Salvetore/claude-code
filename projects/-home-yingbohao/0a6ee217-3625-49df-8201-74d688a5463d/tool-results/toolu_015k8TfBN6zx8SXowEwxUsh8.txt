    61→    61→class MoELoss(torch.nn.Module):
    62→    62→    def __init__(self, model_args):
    63→    63→        super().__init__()
    64→    64→        self.model_args = model_args
    65→    65→        self.tokens_per_expert_buffer = None
    66→    66→        self.sync_group = 'EMPTY'
    67→    67→        self.scale = model_args['moe_expert_num']
    68→    68→        self.mode = model_args['balance_mode']
    69→    69→        assert self.mode in ['global', 'domain', 'subset'], f'Invalid balance mode: {self.mode}'
    70→    70→
    71→    71→    def clear_buffer(self):
    72→    72→        self.tokens_per_expert_buffer = None
    73→    73→
    74→    74→    def set_sync_group(self, sync_group):
    75→    75→        self.sync_group = sync_group
    76→    76→
    77→    77→    def forward(self, all_routing_map, all_expert_prob, tag_index, loss_mask, tag_length):
    78→    78→        if all_routing_map is None or all_expert_prob is None:
    79→    79→            return None
    80→    80→
    81→    81→        # [L, N, E], where L is sequence length, N is token, E is number of experts
    82→    82→        all_routing_map = all_routing_map.clone() * loss_mask.unsqueeze(-1)
    83→    83→        if self.mode in ('global', 'subset'):
    84→    84→            if self.mode == 'subset':
    85→    85→                tag_mask = tag_index != -1
    86→    86→                all_routing_map = all_routing_map * tag_mask.unsqueeze(-1)
    87→    87→                all_expert_prob = all_expert_prob * tag_mask.unsqueeze(-1)
    88→    88→                # tag_index = tag_index * tag_mask
    89→    89→            all_tokens_per_expert = all_routing_map.sum(dim=1, keepdim=True)
    90→    90→        elif self.mode == 'domain':
    91→    91→            all_tokens_per_expert = torch.zeros(all_routing_map.shape[0], tag_length, all_routing_map.shape[2], device=all_routing_map.device, dtype=torch.float32)
    92→    92→            all_tokens_per_expert.scatter_add_(dim=1, index=tag_index[None, :, None].repeat(all_routing_map.shape[0], 1, all_routing_map.shape[2]), src=all_routing_map.float())
    93→    93→
    94→    94→        if self.sync_group != 'EMPTY':
    95→    95→            torch.distributed.all_reduce(all_tokens_per_expert, group=self.sync_group, op=torch.distributed.ReduceOp.AVG)
    96→    96→
    97→    97→        if self.tokens_per_expert_buffer is None:
    98→    98→            self.tokens_per_expert_buffer = all_tokens_per_expert
    99→    99→        else:
   100→   100→            self.tokens_per_expert_buffer = self.tokens_per_expert_buffer + all_tokens_per_expert
   101→   101→        normalized_tokens_per_expert = self.tokens_per_expert_buffer / (self.tokens_per_expert_buffer.sum(dim=-1, keepdim=True) + 1e-6)
   102→   102→        l_aux = torch.bmm(all_expert_prob.float(), normalized_tokens_per_expert.transpose(-1, -2))
   103→   103→        # [L, N, E] * [L, E, K] -> [L, N, K]
   104→   104→        if self.mode == 'domain':
   105→   105→            l_aux = torch.gather(l_aux, dim=2, index=tag_index[None, :, None].repeat(l_aux.shape[0], 1, 1))
   106→   106→        l_aux = self.scale * l_aux.mean(dim=0).flatten()
   107→   107→        l_aux = torch.sum(l_aux * loss_mask)
   108→   108→        return l_aux

   111→   111→class BackBone(torch.nn.Module):
   112→   112→    def __init__(self, model_args, xentropy_chunk_size, xentropy_recompute, precision):
   113→   113→        super().__init__()
   114→   114→        # model weights are fp32 if loading from hf ckpts
   115→   115→        self.model_args = model_args
   116→   116→        self.model = Model(ModelArgs(**model_args))
   117→   117→        self.xentropy_chunk_size = xentropy_chunk_size
   118→   118→        self.xentropy_recompute = xentropy_recompute
   119→   119→        self.precision = precision
   120→   120→        self.ignore_index = model_args['vocab_size'] - 1
   121→   121→        self.fp32_param = ['gate.weight']
   122→   122→        self.fp32_buffer = ['cos_sin_cache']
   123→   123→        if self.precision == 'bf16':
   124→   124→            _logger.info('Casting model parameters to bf16 after optimizer initialization')
   125→   125→            param_dtype, buffer_dtype = torch.bfloat16, torch.bfloat16
   126→   126→            # the name `gate.weight`` is replaced to `gate_weight` in nnscaler
   127→   127→            for name, param in self.model.named_parameters():
   128→   128→                is_fp32_weight = any(fp32_name in name for fp32_name in self.fp32_param)
   129→   129→                if not is_fp32_weight and param.is_floating_point() and param.dtype != torch.bfloat16:
   130→   130→                    _logger.info(f'casting {name} to {param_dtype}')
   131→   131→                    param.data = param.data.to(param_dtype)
   132→   132→                else:
   133→   133→                    _logger.info(f'skipping casting for {name}')
   134→   134→            for name, buf in self.model.named_buffers():
   135→   135→                is_fp32_buffer = False
   136→   136→                for fp32_name in self.fp32_buffer:
   137→   137→                    if fp32_name in name:
   138→   138→                        is_fp32_buffer = True
   139→   139→                        break
   140→   140→                if not is_fp32_buffer and buf.is_floating_point():
   141→   141→                    _logger.info(f'casting {name} to {buffer_dtype}')
   142→   142→                    buf.data = buf.data.to(buffer_dtype)
   143→   143→                else:
   144→   144→                    _logger.info(f'skipping casting for {name}')
   145→   145→
   146→   146→        _logger.info(f'Model: {self.model}')
   147→   147→
   148→   148→    def forward(self, samples):
   149→   149→        tokens, context, targets = samples['tokens'], samples['context'], samples['targets']
   150→   150→        hidden_states, all_routing_map, all_expert_prob = self.model(
   151→   151→            tokens=tokens,
   152→   152→            context=context,
   153→   153→            last_hidden_only=True,
   154→   154→        )
   155→   155→        losses, z_losses = inner_chunk_linear_cross_entropy(hidden_states, self.model.output.weight, targets, self.xentropy_chunk_size, self.xentropy_recompute)
   156→   156→        return losses, z_losses, all_routing_map, all_expert_prob, tokens.size(0)

   159→   159→class WrapperModel(torch.nn.Module):
   160→   160→    def __init__(self, model_args, xentropy_chunk_size, xentropy_recompute, precision):
   161→   161→        super().__init__()
   162→   162→        self.model_args = model_args
   163→   163→        self.backbone = BackBone(model_args, xentropy_chunk_size, xentropy_recompute, precision)
   164→   164→        self.moe_loss = MoELoss(model_args) if model_args['moe'] else None
   165→   165→        self.moe_loss_scale = model_args['moe_loss_scale']
   166→   166→        self.z_loss_scale = model_args['z_loss_scale']
   167→   167→
   168→   168→    def forward(self, samples):
   169→   169→        losses, z_losses, all_routing_map, all_expert_prob, batch_size = self.backbone(samples)
   170→   170→        loss_mask, loss_index, tag_index, dataset_length, tag_length = samples['loss_mask'], samples['loss_index'], samples['tag_index'], samples['dataset_length'], samples['tag_length']
   171→   171→        with torch.no_grad():
   172→   172→            per_source_loss = torch.bincount(loss_index, losses * loss_mask, dataset_length)
   173→   173→            per_source_n_tokens = torch.bincount(loss_index, loss_mask, dataset_length)
   174→   174→        loss = torch.sum(losses * loss_mask)
   175→   175→        z_loss = torch.sum(z_losses * loss_mask)
   176→   176→        n_tokens = loss_mask.sum()
   177→   177→        lm_loss = loss.data
   178→   178→        loss = loss + self.z_loss_scale * z_loss
   179→   179→        if self.model_args['moe']:
   180→   180→            aux_loss = self.moe_loss(all_routing_map, all_expert_prob, tag_index, loss_mask, tag_length)
   181→   181→            moe_loss = aux_loss.data
   182→   182→            loss = loss + self.moe_loss_scale * aux_loss
   183→   183→        else:
   184→   184→            moe_loss = None
   185→   185→        return loss, lm_loss, z_loss, n_tokens, batch_size, moe_loss, per_source_loss, per_source_n_tokens

   188→   188→def aggregate_outputs_fn(loss_outputs, sync_group) -> AggregatedOutputs:
   189→   189→    losses, z_losses, ntokens_info = [], [], []
   190→   190→    per_source_losses, per_source_n_tokens_info = [], []
   191→   191→    moe_losses = []
   192→   192→    for _, loss, z_loss, ntokens, _, moe_loss, per_source_loss, per_source_n_tokens in loss_outputs:
   193→   193→        losses.append(loss)
   194→   194→        z_losses.append(z_loss)
   195→   195→        ntokens_info.append(ntokens)
   196→   196→        moe_losses.append(moe_loss)
   197→   197→        per_source_losses.append(per_source_loss)
   198→   198→        per_source_n_tokens_info.append(per_source_n_tokens)
   199→   199→
   200→   200→    loss_sum = torch.sum(torch.stack(losses), dtype=torch.float64)
   201→   201→    torch.distributed.all_reduce(loss_sum, group=sync_group)
   202→   202→    z_loss_sum = torch.sum(torch.stack(z_losses), dtype=torch.float64)
   203→   203→    torch.distributed.all_reduce(z_loss_sum, group=sync_group)
   204→   204→    ntokens_sum = torch.sum(torch.tensor(ntokens_info, dtype=torch.float64, device=torch.cuda.current_device()))
   205→   205→    torch.distributed.all_reduce(ntokens_sum, group=sync_group)
   206→   206→    per_source_loss_sum = torch.sum(torch.stack(per_source_losses, dim=0), dim=0)
   207→   207→    torch.distributed.all_reduce(per_source_loss_sum, group=sync_group)
   208→   208→    per_source_n_tokens_sum = torch.sum(torch.stack(per_source_n_tokens_info, dim=0), dim=0)
   209→   209→    torch.distributed.all_reduce(per_source_n_tokens_sum, group=sync_group)
   210→   210→    num_batches = torch.tensor(len(losses), device=torch.cuda.current_device())
   211→   211→    torch.distributed.all_reduce(num_batches, group=sync_group)
   212→   212→    if moe_losses[0] is not None:
   213→   213→        moe_loss_sum = torch.sum(torch.stack(moe_losses), dtype=torch.float64)
   214→   214→        torch.distributed.all_reduce(moe_loss_sum, group=sync_group)
   215→   215→        moe_loss_sum = moe_loss_sum.item() / ntokens_sum.item()
   216→   216→    else:
   217→   217→        moe_loss_sum = None
   218→   218→
   219→   219→    per_source_losses = per_source_loss_sum / per_source_n_tokens_sum / math.log(2)
   220→   220→    per_source_losses = torch.where(per_source_n_tokens_sum > 0, per_source_losses, torch.zeros_like(per_source_losses))
   221→   221→    aggregated_outputs = {
   222→   222→        'z_loss': z_loss_sum.item() / ntokens_sum.item(),
   223→   223→        'per_source_losses': per_source_losses.tolist(),
   224→   224→        'per_source_n_tokens': per_source_n_tokens_sum.tolist(),
   225→   225→    }
   226→   226→    if moe_loss_sum is not None:
   227→   227→        aggregated_outputs['moe_loss'] = moe_loss_sum
   228→   228→    return AggregatedOutputs(
   229→   229→        loss_sum=loss_sum.item() / ntokens_sum.item() / math.log(2),
   230→   230→        num_batches=num_batches.item(),
   231→   231→        num_tokens=ntokens_sum.item(),
   232→   232→        aggregated_outputs=aggregated_outputs,
   233→   233→    )

   675→   675→    flops = None

   677→   677→    def on_train_step_start_hook(trainer, batches):
   678→   678→        nonlocal flops
   679→   679→        assert flops is None, f"flops should be None, but got {flops}"
   680→   680→        cur_flops = 0
   681→   681→        for batch in batches:
   682→   682→            cu_seqlens_q = batch['context']['cu_seqlens_q']
   683→   683→            seqlens = cu_seqlens_q[1:] - cu_seqlens_q[:-1]
   684→   684→            # iterate over all seqlens
   685→   685→            for seqlen in seqlens:
   686→   686→                cur_flops += estimate_flops(seqlen.item(), cur_model_args)
   687→   687→        if model_args.dp_sharded:
   688→   688→            sync_group = dist.group.WORLD
   689→   689→        else:
   690→   690→            sync_group = trainer.sync_group
   691→   691→        group_size = torch.distributed.get_world_size(sync_group)
   692→   692→        flops = [None for _ in range(group_size)]
   693→   693→        torch.distributed.all_gather_object(flops, cur_flops / 1e3, group=sync_group)
   694→   694→
   695→   695→        keys = ['tokens', 'targets', 'loss_mask', 'loss_index', 'tag_index']
   696→   696→        start_group = dist.get_rank() // args.plan_ngpus * args.plan_ngpus
   697→   697→        allgather_group = DeviceGroup().get_group(list(range(start_group, start_group + args.plan_ngpus)))
   698→   698→        def allgather_func(sub_content):
   699→   699→            assert sub_content.dim() == 1, f'only support 1D tensor, but got {sub_content.shape}'
   700→   700→            full_size = sub_content.size(0) * args.plan_ngpus
   701→   701→            full_content = torch.empty(full_size, dtype=sub_content.dtype, device=sub_content.device)
   702→   702→            torch.distributed.all_gather_into_tensor(full_content, sub_content, group=allgather_group)
   703→   703→            return full_content
   704→   704→
   705→   705→        if model_args.dp_sharded:
   706→   706→            for i in range(len(batches)):
   707→   707→                batch = batches[i]
   708→   708→                for key in keys:
   709→   709→                    if key not in batch:
   710→   710→                        continue
   711→   711→                    sub_content = batch[key]
   712→   712→                    batches[i][key] = allgather_func(sub_content)
   713→   713→                sub_position = batch['context']['positions']
   714→   714→                batches[i]['context']['positions'] = allgather_func(sub_position)
   715→   715→        elif args.plan_ngpus > 1 and args.check_dataloader_consistency:
   716→   716→            def checker(sub_content):
   717→   717→                full_content = allgather_func(sub_content)
   718→   718→                full_content = full_content.reshape(args.plan_ngpus, -1)
   719→   719→                for i in range(1, args.plan_ngpus):
   720→   720→                    if not torch.equal(full_content[0], full_content[i]):
   721→   721→                        return False
   722→   722→                return True
   723→   723→
   724→   724→            for i in range(len(batches)):
   725→   725→                batch = batches[i]
   726→   726→                for key in keys:
   727→   727→                    if key not in batch:
   728→   728→                        continue
   729→   729→                    if not checker(batch[key]):
   730→   730→                        raise RuntimeError(f'Dataloader inconsistency detected in key {key} at rank {dist.get_rank()}')
   731→   731→
   732→   732→        if trainer.model.moe_loss is not None:
   733→   733→            trainer.model.moe_loss.clear_buffer()
   734→   734→
   735→   735→        z_loss_start_step = trainer.model.model_args['z_loss_start_step']
   736→   736→        z_loss_warmup_step = trainer.model.model_args['z_loss_warmup_step']
   737→   737→        z_loss_target = trainer.model.model_args['z_loss_scale']
   738→   738→        current_step = trainer.train_status.finished_train_steps
   739→   739→        if current_step < z_loss_start_step:
   740→   740→            trainer.model.z_loss_scale = 0.0
   741→   741→        elif current_step < z_loss_start_step + z_loss_warmup_step:
   742→   742→            if current_step - z_loss_start_step == 1:
   743→   743→                print(f"Start z_loss warmup at step {current_step}")
   744→   744→            elif current_step - z_loss_start_step + 1 == z_loss_warmup_step:
   745→   745→                print(f"End z_loss warmup at step {current_step}")
   746→   746→            # Linear warmup from 0 to z_loss_target
   747→   747→            warmup_progress = (current_step - z_loss_start_step) / z_loss_warmup_step
   748→   748→            trainer.model.z_loss_scale = z_loss_target * warmup_progress
   749→   749→        else:
   750→   750→            trainer.model.z_loss_scale = z_loss_target
   751→   751→
   752→   752→        skip_diagnostic = (trainer_args.diagnostic_interval == -1) or (trainer.train_status.finished_train_steps == 0) or ((trainer.train_status.finished_train_steps + 1) % trainer_args.diagnostic_interval != 0)
   753→   753→        nnscaler_funcs.set_skip_diagnostic(skip_diagnostic)

   755→   755→    def on_train_step_end_hook(trainer, losses):
   756→   756→        if trainer.model.moe_loss is not None:
   757→   757→            dense_layers = model_args.dense_layers
