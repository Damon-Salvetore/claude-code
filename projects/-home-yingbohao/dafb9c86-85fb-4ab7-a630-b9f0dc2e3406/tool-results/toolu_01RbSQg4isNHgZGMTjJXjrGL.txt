     4→import math
     5→from torch.optim.lr_scheduler import LRScheduler, Optimizer, _warn_get_lr_called_within_step
     6→
     7→
     8→class WarmupCosineAnnealingLR(LRScheduler):
     9→    r"""
    10→    torch.optim.lr_scheduler.CosineAnnealingLR with warmup.
    11→
    12→    Args:
    13→        optimizer (Optimizer): Wrapped optimizer.
    14→        warmup_steps (int): Number of warmup steps.
    15→        T_max (int): Maximum number of iterations.
    16→        eta_min (float): Minimum learning rate. Default: 0.
    17→        last_epoch (int): The index of last epoch. Default: -1.
    18→        verbose (bool | str): If ``True``, prints a message to stdout for
    19→            each update. Default: ``False``.
    20→
    21→            .. deprecated:: 2.2
    22→                ``verbose`` is deprecated. Please use ``get_last_lr()`` to access the
    23→                learning rate.
    24→    """
    25→
    26→    def __init__(
    27→        self,
    28→        optimizer: Optimizer,
    29→        warmup_steps: int,
    30→        T_max: int,
    31→        eta_min=0.0,
    32→        last_epoch=-1,
    33→    ):  # noqa: D107
    34→        self.warmup_steps = warmup_steps
    35→        self.T_max = T_max - warmup_steps + 1
    36→        self.eta_min = eta_min
    37→        super().__init__(optimizer, last_epoch)
    38→
    39→    def get_lr(self):
    40→        """Retrieve the learning rate of each parameter group."""
    41→        _warn_get_lr_called_within_step(self)
    42→
    43→        last_epoch_wo_warmup = self.last_epoch - self.warmup_steps + 1
    44→        if last_epoch_wo_warmup < 0:
    45→            return [base_lr * (self.last_epoch + 1) / self.warmup_steps for base_lr in self.base_lrs]
    46→        elif last_epoch_wo_warmup == 0:
    47→            return [base_lr  for base_lr in self.base_lrs]
    48→        elif self._step_count == 1 and last_epoch_wo_warmup > 0:
    49→            return [
    50→                self.eta_min
    51→                + (base_lr - self.eta_min)
    52→                * (1 + math.cos((last_epoch_wo_warmup) * math.pi / self.T_max))
    53→                / 2
    54→                for base_lr, group in zip(self.base_lrs, self.optimizer.param_groups)
    55→            ]
    56→        elif (last_epoch_wo_warmup - 1 - self.T_max) % (2 * self.T_max) == 0:
    57→            return [
    58→                group["lr"]
    59→                + (base_lr - self.eta_min) * (1 - math.cos(math.pi / self.T_max)) / 2
    60→                for base_lr, group in zip(self.base_lrs, self.optimizer.param_groups)
    61→            ]
    62→        return [
    63→            (1 + math.cos(math.pi * last_epoch_wo_warmup / self.T_max))
    64→            / (1 + math.cos(math.pi * (last_epoch_wo_warmup - 1) / self.T_max))
    65→            * (group["lr"] - self.eta_min)
    66→            + self.eta_min
    67→            for group in self.optimizer.param_groups
    68→        ]
    69→
    70→    def _get_closed_form_lr(self):
    71→        last_epoch_wo_warmup = self.last_epoch - self.warmup_steps + 1
    72→        if last_epoch_wo_warmup < 0:
    73→            return [base_lr * (self.last_epoch + 1) / self.warmup_steps for base_lr in self.base_lrs]
    74→        else:
    75→            return [
    76→                self.eta_min
    77→                + (base_lr - self.eta_min)
    78→                * (1 + math.cos(math.pi * last_epoch_wo_warmup / self.T_max))
    79→                / 2
    80→                for base_lr in self.base_lrs
    81→            ]
