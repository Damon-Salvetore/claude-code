   159→class WrapperModel(torch.nn.Module):
   160→    def __init__(self, model_args, xentropy_chunk_size, xentropy_recompute, precision):
   161→        super().__init__()
   162→        self.model_args = model_args
   163→        self.backbone = BackBone(model_args, xentropy_chunk_size, xentropy_recompute, precision)
   164→        self.moe_loss = MoELoss(model_args) if model_args['moe'] else None
   165→        self.moe_loss_scale = model_args['moe_loss_scale']
   166→        self.z_loss_scale = model_args['z_loss_scale']
   167→
   168→    def forward(self, samples):
   169→        losses, z_losses, all_routing_map, all_expert_prob, batch_size = self.backbone(samples)
   170→        loss_mask, loss_index, tag_index, dataset_length, tag_length = samples['loss_mask'], samples['loss_index'], samples['tag_index'], samples['dataset_length'], samples['tag_length']
   171→        with torch.no_grad():
   172→            per_source_loss = torch.bincount(loss_index, losses * loss_mask, dataset_length)
   173→            per_source_n_tokens = torch.bincount(loss_index, loss_mask, dataset_length)
   174→        loss = torch.sum(losses * loss_mask)
   175→        z_loss = torch.sum(z_losses * loss_mask)
   176→        n_tokens = loss_mask.sum()
   177→        lm_loss = loss.data
   178→        loss = loss + self.z_loss_scale * z_loss
   179→        if self.model_args['moe']:
   180→            aux_loss = self.moe_loss(all_routing_map, all_expert_prob, tag_index, loss_mask, tag_length)
   181→            moe_loss = aux_loss.data
   182→            loss = loss + self.moe_loss_scale * aux_loss
   183→        else:
   184→            moe_loss = None
   185→        return loss, lm_loss, z_loss, n_tokens, batch_size, moe_loss, per_source_loss, per_source_n_tokens

   188→def aggregate_outputs_fn(loss_outputs, sync_group) -> AggregatedOutputs:
   189→    losses, z_losses, ntokens_info = [], [], []
   190→    per_source_losses, per_source_n_tokens_info = [], []
   191→    moe_losses = []
   192→    for _, loss, z_loss, ntokens, _, moe_loss, per_source_loss, per_source_n_tokens in loss_outputs:
   193→        losses.append(loss)
   194→        z_losses.append(z_loss)
   195→        ntokens_info.append(ntokens)
   196→        moe_losses.append(moe_loss)
   197→        per_source_losses.append(per_source_loss)
   198→        per_source_n_tokens_info.append(per_source_n_tokens)
   199→
   200→    loss_sum = torch.sum(torch.stack(losses), dtype=torch.float64)
   201→    torch.distributed.all_reduce(loss_sum, group=sync_group)
   202→    z_loss_sum = torch.sum(torch.stack(z_losses), dtype=torch.float64)
   203→    torch.distributed.all_reduce(z_loss_sum, group=sync_group)
   204→    ntokens_sum = torch.sum(torch.tensor(ntokens_info, dtype=torch.float64, device=torch.cuda.current_device()))
   205→    torch.distributed.all_reduce(ntokens_sum, group=sync_group)
   206→    per_source_loss_sum = torch.sum(torch.stack(per_source_losses, dim=0), dim=0)
   207→    torch.distributed.all_reduce(per_source_loss_sum, group=sync_group)
   208→    per_source_n_tokens_sum = torch.sum(torch.stack(per_source_n_tokens_info, dim=0), dim=0)
   209→    torch.distributed.all_reduce(per_source_n_tokens_sum, group=sync_group)
   210→    num_batches = torch.tensor(len(losses), device=torch.cuda.current_device())
   211→    torch.distributed.all_reduce(num_batches, group=sync_group)
   212→    if moe_losses[0] is not None:
   213→        moe_loss_sum = torch.sum(torch.stack(moe_losses), dtype=torch.float64)
   214→        torch.distributed.all_reduce(moe_loss_sum, group=sync_group)
   215→        moe_loss_sum = moe_loss_sum.item() / ntokens_sum.item()
   216→    else:
   217→        moe_loss_sum = None
   218→
   219→    per_source_losses = per_source_loss_sum / per_source_n_tokens_sum / math.log(2)
   220→    per_source_losses = torch.where(per_source_n_tokens_sum > 0, per_source_losses, torch.zeros_like(per_source_losses))
   221→    aggregated_outputs = {
   222→        'z_loss': z_loss_sum.item() / ntokens_sum.item(),
   223→        'per_source_losses': per_source_losses.tolist(),
   224→        'per_source_n_tokens': per_source_n_tokens_sum.tolist(),
   225→    }
   226→    if moe_loss_sum is not None:
   227→        aggregated_outputs['moe_loss'] = moe_loss_sum
   228→    return AggregatedOutputs(
   229→        loss_sum=loss_sum.item() / ntokens_sum.item() / math.log(2),
   230→        num_batches=num_batches.item(),
   231→        num_tokens=ntokens_sum.item(),
   232→        aggregated_outputs=aggregated_outputs,
   233→    )
