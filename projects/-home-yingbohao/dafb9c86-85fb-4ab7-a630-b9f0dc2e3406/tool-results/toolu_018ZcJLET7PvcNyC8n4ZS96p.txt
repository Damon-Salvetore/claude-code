    61→class MoELoss(torch.nn.Module):
    62→    def __init__(self, model_args):
    63→        super().__init__()
    64→        self.model_args = model_args
    65→        self.tokens_per_expert_buffer = None
    66→        self.sync_group = 'EMPTY'
    67→        self.scale = model_args['moe_expert_num']
    68→        self.mode = model_args['balance_mode']
    69→        assert self.mode in ['global', 'domain', 'subset'], f'Invalid balance mode: {self.mode}'
    70→
    71→    def clear_buffer(self):
    72→        self.tokens_per_expert_buffer = None
    73→
    74→    def set_sync_group(self, sync_group):
    75→        self.sync_group = sync_group
    76→
    77→    def forward(self, all_routing_map, all_expert_prob, tag_index, loss_mask, tag_length):
    78→        if all_routing_map is None or all_expert_prob is None:
    79→            return None
    80→
    81→        # [L, N, E], where L is sequence length, N is token, E is number of experts
    82→        all_routing_map = all_routing_map.clone() * loss_mask.unsqueeze(-1)
    83→        if self.mode in ('global', 'subset'):
    84→            if self.mode == 'subset':
    85→                tag_mask = tag_index != -1
    86→                all_routing_map = all_routing_map * tag_mask.unsqueeze(-1)
    87→                all_expert_prob = all_expert_prob * tag_mask.unsqueeze(-1)
    88→                # tag_index = tag_index * tag_mask
    89→            all_tokens_per_expert = all_routing_map.sum(dim=1, keepdim=True)
    90→        elif self.mode == 'domain':
    91→            all_tokens_per_expert = torch.zeros(all_routing_map.shape[0], tag_length, all_routing_map.shape[2], device=all_routing_map.device, dtype=torch.float32)
    92→            all_tokens_per_expert.scatter_add_(dim=1, index=tag_index[None, :, None].repeat(all_routing_map.shape[0], 1, all_routing_map.shape[2]), src=all_routing_map.float())
    93→
    94→        if self.sync_group != 'EMPTY':
    95→            torch.distributed.all_reduce(all_tokens_per_expert, group=self.sync_group, op=torch.distributed.ReduceOp.AVG)
    96→
    97→        if self.tokens_per_expert_buffer is None:
    98→            self.tokens_per_expert_buffer = all_tokens_per_expert
    99→        else:
   100→            self.tokens_per_expert_buffer = self.tokens_per_expert_buffer + all_tokens_per_expert
   101→        normalized_tokens_per_expert = self.tokens_per_expert_buffer / (self.tokens_per_expert_buffer.sum(dim=-1, keepdim=True) + 1e-6)
   102→        l_aux = torch.bmm(all_expert_prob.float(), normalized_tokens_per_expert.transpose(-1, -2))
   103→        # [L, N, E] * [L, E, K] -> [L, N, K]
   104→        if self.mode == 'domain':
   105→            l_aux = torch.gather(l_aux, dim=2, index=tag_index[None, :, None].repeat(l_aux.shape[0], 1, 1))
   106→        l_aux = self.scale * l_aux.mean(dim=0).flatten()
   107→        l_aux = torch.sum(l_aux * loss_mask)
   108→        return l_aux
