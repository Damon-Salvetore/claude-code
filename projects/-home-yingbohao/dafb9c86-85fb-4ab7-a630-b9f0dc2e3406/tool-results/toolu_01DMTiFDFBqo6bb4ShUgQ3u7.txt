   111→def parse_training_args():
   112→    parser = argparse.ArgumentParser(description="training arguments")
   113→    parser.add_argument(f"--model", type=str, choices=model_args.keys())
   114→    parser.add_argument(f"--data", type=str, choices=data_args.keys())
   115→    parser.add_argument(f"--hyperparams", type=str, choices=training_args.keys())
   116→
   117→    argument_dict = {}
   118→
   119→    # Add arguments for each field in the dataclass
   120→    for config_type in [ModelArgs, DataLoaderArgs, TrainerArgs]:
   121→        for field in fields(config_type):
   122→            if field.name in argument_dict:
   123→                continue
   124→
   125→            argument_dict[field.name] = True
   126→            field_type = field.type
   127→            kwargs = {}
   128→
   129→            if field_type == bool:
   130→                kwargs["action"] = "store_true"
   131→                parser.add_argument(f"--{field.name}", **kwargs)
   132→            else:
   133→                kwargs["default"] = None
   134→                parser.add_argument(f"--{field.name}", type=field_type, **kwargs)
   135→
   136→
   137→    custom_args = parser.parse_args()
   138→
   139→    _model_args = copy.deepcopy(model_args[custom_args.model])
   140→    _data_args = copy.deepcopy(data_args[custom_args.data])
   141→    _trainer_args = copy.deepcopy(training_args[custom_args.hyperparams])
   142→
   143→    for k, v in custom_args.__dict__.items():
   144→        if v is not None and v is not False:
   145→            _model_args.__dict__[k] = v
   146→            _data_args.__dict__[k] = v
   147→            _trainer_args.__dict__[k] = v
   148→
   149→    return _model_args, _data_args, _trainer_args
